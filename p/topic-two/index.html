const ll N=2e5+10;
ll n,m;
ll a[N],b[N];

    for(ll i=0;i<n;i++) cin>>b[i];
    ll Left=0,Right=1e7;

        ll mid=(Right+Left+1)/2;
        if(Check(mid)) Left=mid;
        else Right=mid-1;

    ll sum=0;

        if(x-a[i]>b[i]) return 0;

    }
    return sum<=m;
}

    cin>>book>>people;

        cin>>books[i];
        right+=books[i];
        left=max(left,books[i]);

#include<iostream>
#include<climits>
#define ll long long
using namespace std;

ll Check(ll y);

const ll N=1e5+1;

ll width,days;
ll H[N],Sum[N]={0};

int main() {
    cin>>width>>days;
    days*=2;
    for(ll i=1;i<width;i++) {
        cin>>H[i];
        Sum[i]=Sum[i-1]+H[i];
    }
    ll left=1,right=width;
    while(left<right) {
        ll mid=(left+right)/2;
        if(Check(mid)) right=mid;
        else left=mid+1;
    }
    cout<<left<<endl;
    return 0;
}

ll Check(ll y) {
    for(ll i=y;i<width;i++) {
        if(Sum[i]-Sum[i-y]<days) return 0;
    }
    return 1;
}
